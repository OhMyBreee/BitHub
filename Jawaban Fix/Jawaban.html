            #include <stdio.h>
            #include <string.h>
            int main(){
                char str[105];
                scanf("%[^\n]", str); getchar();
            
                int length = strlen(str);
            
                // reverse string
                for (int i = 0; i < length / 2; i++){
                    char temp = str[i];
                    str[i] = str[length - 1 - i];
                    str[length - 1 - i] = temp;
                }
            
                for (int i = 0; i < length; i++){
                    if (str[i] >= 'A' && str[i] <= 'Z'){
                        // ubah uppercase ke lowercase
                        str[i] += 32;
                    } else if (str[i] >= 'a' && str[i] <= 'z'){
                        // ubah lowercase ke uppercase
                        str[i] -= 32;
                    }
                }
            
                printf ("%s\n", str);
            
                return 0;
            }
